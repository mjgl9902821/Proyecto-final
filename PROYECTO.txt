Create table rol(
	id int IDENTITY(1,1) PRIMARY KEY,
	Nombre varchar(55),
);

Create table usuario(
	id int IDENTITY(1,1) PRIMARY KEY,
	Nombres varchar(255),
	Apellidos varchar(255),
	Telefono varchar(255),
	Direccion varchar(255),
	Correo varchar(255),
	Posición varchar(255),
	Rol_id int foreign key references rol(id)
);

	id int IDENTITY(1,1) PRIMARY KEY,
	Nombre varchar(255),
	Descripcion varchar(255)
);
